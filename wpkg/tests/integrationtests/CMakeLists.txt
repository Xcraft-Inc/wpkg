project(integrationtest)

include_directories(
    "${PROJECT_BINARY_DIR}"
    "${PROJECT_SOURCE_DIR}"
)

add_executable( ${PROJECT_NAME}
    integrationtest_main.cpp
    integrationtest_package.cpp
    integrationtest_system.cpp
    ${CMAKE_SOURCE_DIR}/tools/license.cpp
)

StaticCompile()

if( MSVC )
    set_target_properties( ${PROJECT_NAME} PROPERTIES FOLDER ${TEST_FOLDER} )
endif()

target_link_libraries( ${PROJECT_NAME}
    tests_common
    debpackages_static
    expr_static
    wpkg_z_static
    wpkg_bz2_static
)

unset( DEBUG_WORD )
if( "${CMAKE_BUILD_TYPE}" STREQUAL "Debug" )
    set( DEBUG_WORD "_d" )
endif()

add_test( integrationtests ${PROJECT_NAME}
    --tmp ${TMPDIR}
    --wpkg ${CMAKE_BINARY_DIR}/tools/wpkg_static${DEBUG_WORD}${CMAKE_EXECUTABLE_SUFFIX}
)
